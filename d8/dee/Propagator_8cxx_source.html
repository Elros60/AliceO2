38 bool Propagator::PropagateToXBxByBz(TrackParCov& track, float xToGo, float mass, float maxSnp, float maxStep,
39  int matCorr, int signCorr)
40 {
41  //----------------------------------------------------------------
42  //
43  // Propagates the track to the plane X=xk (cm)
44  // taking into account all the three components of the magnetic field
45  // and correcting for the crossed material.
46  //
47  // mass - mass used in propagation - used for energy loss correction (if <0 then q=2)
48  // maxStep - maximal step for propagation
49  //
50  //----------------------------------------------------------------
51  const float Epsilon = 0.00001;
52  auto dx = xToGo - track.getX();
53  int dir = dx > 0.f ? 1 : -1;
54  if (!signCorr) {
55  signCorr = -dir; // sign of eloss correction is not imposed
56  }
57 
58  std::array<float, 3> b;
59  while (std::abs(dx) > Epsilon) {
60  auto step = std::min(std::abs(dx), maxStep);
61  if (dir < 0) {
62  step = -step;
63  }
64  auto x = track.getX() + step;
65  auto xyz0 = track.getXYZGlo();
66  mField->Field(xyz0, b.data());
67 
68  if (!track.propagateTo(x, b))
69  return false;
70  if (maxSnp > 0 && std::abs(track.getSnp()) >= maxSnp)
71  return false;
72 
73  if (matCorr) {
74  auto xyz1 = track.getXYZGlo();
75  auto mb = GeometryManager::MeanMaterialBudget(xyz0, xyz1);
76  if (signCorr < 0) {
77  mb.length = -mb.length;
78  }
79  //
80  if (!track.correctForMaterial(mb.meanX2X0, mb.meanRho * mb.length, mass))
81  return false;
82  }
83  dx = xToGo - track.getX();
84  }
85  return true;
86 }
